                    function randomR() {
  function f(r) {
    const a = 0.529;
    return (4 / (a * a * a)) * r * r * Math.exp((-2 * r) / a);
  }
  const m = 1.023;
  do {
    var x = randf(0, 3),
      y = randf(0, m);
    if (y <= f(x)) return x;
  } while (true);
}
function randomPoint() {
  var r = randomR();
  var phi = randf(0, 2 * Math.PI);
  var theta = Math.acos(randf(-1, 1));
  return [r, theta, phi];
}
function spherical(r, theta, phi) {
  return (
    r *
    [
      Math.sin(theta) * Math.cos(phi),
      Math.sin(theta) * Math.sin(phi),
      Math.cos(theta),
    ]
  );
}
